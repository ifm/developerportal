171  template <typename T>
172  T
173  mkval(const unsigned char* buff)
174  {
175  union
176  {
177  T v;
178  unsigned char bytes[sizeof(T)];
179  } value;
180 
181 #if !defined(_WIN32) && __BYTE_ORDER == __BIG_ENDIAN
182  std::reverse_copy(buff, buff + sizeof(T), value.bytes);
183 #else
184  std::copy(buff, buff + sizeof(T), value.bytes);
185 #endif
186 
187  return value.v;
188  }
